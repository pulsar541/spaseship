package com.mygdx.game;

import java.util.ArrayList;
import java.util.List;

import kepler.ContactListener;
import kepler.Vec3;
import kepler.World;

public class BallsContactListener extends ContactListener {

    List<Integer> memIndexes = new ArrayList<>();

    public BallsContactListener(World world) {
        super(world);
    }

    @Override
    public void beginContactWall(int bodyIndex, int bodyUid, World.Walls wall) {
        Ball ball = (Ball)world.getBody(bodyIndex);
        ball.isNew = false;
        if(ball.neighboursIndexes.size() >= 3) {
            if (ball.getSpeed() < ball.getRadius() * 0.01f) {
                ball.setKinematic(true);
            }
        }
    }

    @Override
    public void beginContactBodies(int firstBodyIndex, int secondBodyIndex, Vec3 point) {
        Ball firstBall = (Ball)world.getBody(firstBodyIndex);
        Ball secondBall = (Ball)world.getBody(secondBodyIndex);

        firstBall.isNew = false;
        secondBall.isNew = false;

        if(firstBall.neighboursIndexes.size() >= 5) {
            if (firstBall.getSpeed() < firstBall.getRadius() * 0.01f)
                firstBall.setKinematic(true);
            else    if(firstBall.neighboursIndexes.size() >= 6) {
                firstBall.setKinematic(true);
            }
        }


        if(secondBall.neighboursIndexes.size() >= 5) {
            if (secondBall.getSpeed() < firstBall.getRadius() * 0.01f)
                secondBall.setKinematic(true);
            else    if(secondBall.neighboursIndexes.size() >= 6) {
                secondBall.setKinematic(true);
            }
        }


        if(secondBall.isKinematic && secondBall.pos.y > firstBall.pos.y ) {
            firstBall.setKinematic(true);
        }
        if(firstBall.isKinematic && firstBall.pos.y > secondBall.pos.y ) {
            secondBall.setKinematic(true);
        }



        if(firstBall.neighboursIndexes.size() >= 2) {
            for(Integer item : firstBall.neighboursIndexes){
                Ball nb = ((Ball)world.getBody(item.intValue()));
                if(nb.type() == firstBall.type()) {
                    memIndexes.add(item);

                    if(memIndexes.size() >= 2) {
                        firstBall.life = 0;
                        for(Integer itemMem : memIndexes){
                            Ball nnb = ((Ball)world.getBody(itemMem.intValue()));
                            nnb.life = 0;
                        }
                        memIndexes.clear();

                    }

                }
            }


        }

    }
}
